Jaika v2 Hosting Guide

Deploy Jaika v2 on a fresh Ubuntu/Debian server. Powered by Google's AI APIs with OAuth authentication.

1. Install system dependencies

sudo apt update
sudo apt install -y python3 python3-pip python3-venv curl git \
  pandoc texlive-latex-base texlive-latex-extra texlive-fonts-recommended

python3 --version  # Should be 3.11+

2. Clone/copy Jaika v2

sudo mkdir -p /opt/jaika-v2
sudo git clone https://github.com/goyaljai/jaika.git /opt/jaika-v2
# Or copy: sudo cp -r /path/to/jaika-v2/* /opt/jaika-v2/

3. Python venv + dependencies

cd /opt/jaika-v2
python3 -m venv .venv
source .venv/bin/activate
pip install --upgrade pip
pip install -r requirements.txt
# requirements.txt includes: flask, requests, google-auth, google-auth-oauthlib,
# gunicorn, python-dotenv, python-docx, openpyxl, python-pptx

For Android chroot: unset ANDROID_DATA ANDROID_ROOT before pip commands.

4. Create data directories + config

mkdir -p /opt/jaika-v2/data/{users,skills}
mkdir -p /opt/jaika-v2/static
mkdir -p /var/log/server

# Admin emails (goyaljai.y14@gmail.com is always hardcoded admin)
echo '["your-admin@gmail.com"]' > /opt/jaika-v2/data/admins.json
echo '[]' > /opt/jaika-v2/data/pro_users.json

# Generate secret key
SK=$(python3 -c "import secrets; print(secrets.token_hex(32))")

cat > /opt/jaika-v2/.env <<EOF
JAIKA_DATA_DIR=/opt/jaika-v2/data
JAIKA_HOST=0.0.0.0
JAIKA_PORT=5244
SECRET_KEY=$SK
EOF

5. Auth setup — OAuth credentials

# Add OAuth credentials to .env:
echo "GOOGLE_CLIENT_ID=YOUR_CLIENT_ID" >> /opt/jaika-v2/.env
echo "GOOGLE_CLIENT_SECRET=YOUR_CLIENT_SECRET" >> /opt/jaika-v2/.env

# First user login: visit https://YOUR-DOMAIN/ and sign in
# The OAuth flow stores tokens per-user in data/users/<user_id>/token.json

6. Set up Supervisor

sudo apt install -y supervisor

cat > /etc/supervisor/conf.d/jaika.conf <<'EOF'
[program:jaika]
command=/opt/jaika-v2/.venv/bin/gunicorn \
  --bind 0.0.0.0:5244 --workers 2 --threads 4 \
  --timeout 180 --access-logfile - app:app
directory=/opt/jaika-v2
environment=HOME="/root",PATH="/opt/jaika-v2/.venv/bin:/usr/local/bin:/usr/bin:/bin"
autostart=true
autorestart=true
stdout_logfile=/var/log/server/jaika-out.log
stderr_logfile=/var/log/server/jaika-err.log
EOF

supervisorctl reread && supervisorctl update

7. Set up Tailscale (optional, for HTTPS)

curl -fsSL https://tailscale.com/install.sh | sh
tailscale up --ssh

# Expose via HTTPS Funnel
tailscale funnel --bg --https 443 http://localhost:5244
tailscale serve status

Your app will be at https://YOUR-HOSTNAME.tail*.ts.net

8. Cron — cleanup old files

crontab -e
# Add:
0 3 * * * find /opt/jaika-v2/data/users/*/uploads/ /opt/jaika-v2/data/users/*/outputs/ -type f -mmin +120 -delete 2>/dev/null

9. Verify

supervisorctl status

# Test health
curl -s http://localhost:5244/ | head -1

# Test API (replace with real user_id after first login)
curl -s http://localhost:5244/api/docs | python3 -m json.tool | head -20

10. Quick reference

# Restart
supervisorctl restart jaika

# Logs
tail -f /var/log/server/jaika-out.log
tail -f /var/log/server/jaika-err.log

# Update
cd /opt/jaika-v2 && git pull
source .venv/bin/activate && pip install -r requirements.txt
supervisorctl restart jaika

# Multiple users: fully isolated — each user gets their own:
# data/users/<user_id>/{sessions/, uploads/, outputs/, memory.json, token.json}
# Rate limit: each user uses their own Google quota (60 req/min free tier)
